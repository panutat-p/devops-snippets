MacOS
```sh
curl -o 'aws_cli.pkg' 'https://awscli.amazonaws.com/AWSCLIV2.pkg'
open aws_cli.pkg
```

```sh
ln -s $HOME/aws-cli/aws /usr/local/bin/aws
ln -s $HOME/aws-cli/aws_completer /usr/local/bin/aws_completer
```

```sh
which aws
```

```sh
aws --version
```
